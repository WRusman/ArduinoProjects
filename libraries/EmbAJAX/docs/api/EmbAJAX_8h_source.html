551 public:
557  EmbAJAXRadioGroup(const char* id_base, const char* options[NUM], uint8_t selected_option = 0) : EmbAJAXContainer<NUM>(), EmbAJAXRadioGroupBase() {
558  for (uint8_t i = 0; i < NUM; ++i) {
559  char* childid = childids[i];
560  strncpy(childid, id_base, ARDUJAX_MAX_ID_LEN-4);
561  itoa(i, &(childid[strlen(childid)]), 10);
562  buttons[i] = EmbAJAXCheckButton(childid, options[i], i == selected_option);
563  buttons[i].radiogroup = this;
564  buttonpointers[i] = &buttons[i];
565  }
566  _current_option = selected_option;
567  EmbAJAXContainer<NUM>::_children = buttonpointers; // Hm, why do I need to specify EmbAJAXContainer<NUM>::, explicitly?
568  _name = id_base;
569  }
571  void selectOption(uint8_t num) {
572  for (uint8_t i = 0; i < NUM; ++i) {
573  buttons[i].setChecked(i == num);
574  }
575  _current_option = num; // NOTE: might be outside of range, but that's ok, signifies "none selected"
576  }
578  uint8_t selectedOption() const {
579  return _current_option;
580  }
583  EmbAJAXBase* button(uint8_t num) {
584  if (num < NUM) return (&buttons[num]);
585  return 0;
586  }
587 private:
588  EmbAJAXCheckButton buttons[NUM];
589  EmbAJAXBase* buttonpointers[NUM];
590  char childids[NUM][ARDUJAX_MAX_ID_LEN];
591  int8_t _current_option;
592  void selectOption(EmbAJAXCheckButton* which) override {
593  _current_option = -1;
